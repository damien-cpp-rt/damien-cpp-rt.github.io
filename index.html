Highlights

  • 인터리브드 f32(LRLR…) 믹스, 오디오 콜백은 논블로킹 pop + 기본 무음
  • 워커: 트랙 라운드로빈(rr)으로 CHUNK_DECODE 단위 디코딩 → rtrb push
  • 링버퍼: RB1_FRAMES = 16,777,216 (frames)
  • 아키텍처: 디코더 → rtrb 링버퍼(인터리브드) → 오디오 콜백(인터리브드) → VST3 체인
  • 운영 원칙: 계산 단위는 frames. 콜백은 0.0으로 초기화 후 pop된 샘플만 누적.
  • 스케줄링: 매 사이클 rr 시작점을 회전, 각 트랙은 is_full() 아니면 최대 CHUNK_DECODE 프레임 push.

Core Mixing Snippet

// rust_render_interleaved: out[L R L R …]
out.fill(0.0); //무음 설정
for (ti, cons_mx) in eng.consumers.iter().enumerate() { //트랙별 순회
  if ti >= params.volume.len() || ti >= params.pan.len() || ti >= params.muted.len() { continue; } //안전
  let muted = params.muted[ti].load(Ordering::Relaxed); //뮤트 체크
  let vol = f32::from_bits(params.volume[ti].load(Ordering::Relaxed)).clamp(0.0, 1.0); //볼륨
  let pan = f32::from_bits(params.pan[ti].load(Ordering::Relaxed)).clamp(-1.0, 1.0); //팬
  if muted || vol == 0.0 { continue; } //스킵
  let gl = vol * (1.0 - pan) * 0.5; //L
  let gr = vol * (1.0 + pan) * 0.5; //R

  let mut underrun = false; //언더런 플래그
  if let Ok(mut cons) = cons_mx.lock() { //락 획득
    for f in 0..frames { //프레임 루프
      let l = cons.pop().unwrap_or(0.0); //L
      let r = cons.pop().unwrap_or(0.0); //R
      if l == 0.0 && r == 0.0 { underrun = true; } // 언더런 감지(예시)
      let i = f * 2; //인덱스 계산
      unsafe { //성능 위해 언체크드
        *out.get_unchecked_mut(i)     += l * gl; // L
        *out.get_unchecked_mut(i + 1) += r * gr; // R
      }
    }
  }
  if underrun { let (_, cv) = &*eng.thread_wait; cv.notify_all(); } //언더런 시 워커 알림
}
